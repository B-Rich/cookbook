$\cos 0^{\circ} = 1$, 
\par
$\cos 30^{\circ} = \cos \frac{\pi}{6} = \frac{\sqrt{3}}{2}$,
\par
$\cos 45^{\circ} = \cos \frac{\pi}{4} = \frac{1}{\sqrt{2}}$,
\par
$\cos 60^{\circ} = \cos \frac{\pi}{3} = \frac{1}{2}$,
\par
$\cos 90^{\circ} = cos \frac{\pi}{2} = 0$,
\par
$\cos 180^{\circ} = cos \pi = -1$.
